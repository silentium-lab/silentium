Класс SourceApplied

SourceApplied на GitHub

Иногда к значению источника нужно применить какую-нибудь существующую функцию, для этого можно использовать класс SourceApplied

Пример 1.

import { source, SourceApplied, patron } from "silentium";

const guest = v => {
  const r = document.querySelector(".guest-result");
  if (r) {
    r.textContent = v;
  }
}

const max = 10;
let iterations = 0;
const timer = source((g) => {
  const tick = () => {
    iterations++;
    g.give(iterations);
    setTimeout(tick, 1000);
  }
tick();
});

sourceApplied(timer, x => x * 2).value(patron(guest));
