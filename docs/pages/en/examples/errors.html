Example 1.

import { SourceActive, SourceEmpty, guestCast, patron } from "silentium";

  class Random {
    constructor(till) {
      this.activeSource = new SourceActive((_, source) => {
        source.give(Math.round(Math.random() * till));
      });
    }

    do() {
      this.activeSource.do();
      return this;
    }

    value(guest) {
      this.activeSource.value(guest);
      return this;
    }
  }

  window.random = new Random(99);

  class EvenNumberMessage {
    constructor(baseNumber, errorGuest) {
      this.baseNumber = baseNumber;
      this.errorGuest = errorGuest;
    }

    value(guest) {
      this.baseNumber.value(
        guestCast(guest, (value) => {
          if (value % 2 === 0) {
            guest.give(`${value} - even!`);
          } else {
            this.errorGuest.give(`${value} is not even!`);
          }
        })
      );
      return this;
    }
  }

  const randomErrors = new SourceEmpty();
  const randomMessage = new EvenNumberMessage(random, randomErrors);

  random.value(
    patron((val) => {
      document.querySelector(".guest-result").textContent = val;
    })
  );
  randomMessage.value(
    patron((val) => {
      document.querySelector(".guest-message").textContent = val;
    })
  );
  randomErrors.value(
    patron((val) => {
      document.querySelector(".guest-error").textContent = val;
    })
  );

At the application level, if we need to handle errors centrally, we can create one object from the SourceEmpty class and pass this object as a visitor for any errors, which will allow you to collect all errors in one place and process them.
