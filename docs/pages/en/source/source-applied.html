Sometimes you need to apply some existing function to the source value, you can use the SourceApplied for this.

Example 1.

import { source, SourceApplied, patron } from "silentium";

const guest = v => {
  const r = document.querySelector(".guest-result");
  if (r) {
    r.textContent = v;
  }
}

const max = 10;
let iterations = 0;
const timer = source((g) => {
  const tick = () => {
    iterations++;
    g.give(iterations);
    setTimeout(tick, 1000);
  }
tick();
});

sourceApplied(timer, x => x * 2).value(patron(guest));
