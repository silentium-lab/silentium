The SourceMap class is needed when processing multiple values, via The creation of objects based on these values is data sources. For example, we have a class that is the result of multiplying a number by 2. SourceMap allows you to apply this class to some iterable data source, converting each value to a new one. As a result, we get a new iterable data source in which all the numbers are multiplied by 2 .

Example 1.

import { Source, sourceChangeable, guestCast, sourceMap, lazyClass, give } from "silentium";

class X2 {
  constructor(baseNumber) {
    this.baseNumber = baseNumber;
  }

  value(guest) {
    this.baseNumber.value(
      guestCast(guest, (v) => {
        give(v * 2, guest);
      })
    );
    return this;
  }
}

const source = sourceChangeable([1, 2, 3, 9]);
const guestMapped = sourceMap(source, lazyClass(X2));

guestMapped.value((v) => {
  document.querySelector(".guest-result").textContent = v.join();
});

Example 2.

import {
  source,
  sourceChangeable,
  guestCast,
  SourceMap,
  lazyClass,
  give,
  isSource,
} from "silentium";

(() => {
  class X3 {
    constructor(baseNumber) {
      this.baseNumber = baseNumber;
    }

    value(guest) {
      this.baseNumber.value(
        guestCast(guest, (v) => {
          give(v * 3, guest);
        })
      );
      return this;
    }
  }

  const numSource = (num) =>
    source((guest) =>
      setTimeout(() => {
        give(num, guest);
      }, 8000)
    );

  const source = sourceChangeable([
    numSource(1),
    numSource(2),
    numSource(3),
    numSource(4),
    numSource(5),
  ]);
  const map = sourceMap(source, lazyClass(X3));

  map.value((v) => {
    document.querySelector(".guest-result-2").textContent = v.join();
  });
})();
